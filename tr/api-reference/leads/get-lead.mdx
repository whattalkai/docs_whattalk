---
title: "Müşterileri listele"
api: "GET /user/leads"
description: "Kimliği doğrulanmış kullanıcının tüm müşterilerini filtreleme ve sayfalama seçenekleriyle listeleyin"
---

Bu endpoint, kimliği doğrulanmış kullanıcıya ait tüm müşterileri çeşitli filtreleme ve sayfalama seçenekleriyle listelemenizi sağlar.

### Sorgu Parametreleri

<ParamField query="status" type="string" optional>
  Müşterileri duruma göre filtreleyin. Olası değerler: `created`, `scheduled`, `processing`, `completed`, `rescheduled`, `reached-max-retries`, `blacklisted`
</ParamField>

<ParamField query="campaign_id" type="integer" optional>
  Müşterileri kampanya kimliğine göre filtreleyin
</ParamField>

<ParamField query="phone_number" type="string" optional>
  Müşterileri telefon numarasına göre filtreleyin (kısmi eşleşme desteklenir)
</ParamField>

<ParamField query="date_from" type="string" optional>
  Bu tarihten itibaren oluşturulan müşterileri filtreleyin (YYYY-MM-DD formatı)
</ParamField>

<ParamField query="date_to" type="string" optional>
  Bu tarihe kadar oluşturulan müşterileri filtreleyin (YYYY-MM-DD formatı)
</ParamField>

<ParamField query="per_page" type="integer" optional>
  Sayfa başına müşteri sayısı (1-100, varsayılan: 15)
</ParamField>

<ParamField query="page" type="integer" optional>
  Sayfa numarası (varsayılan: 1)
</ParamField>

### Yanıt alanları

<ResponseField name="data" type="array">
  <Expandable title="özellikler">
    <ResponseField name="id" type="integer">
      Müşterinin kimliği
    </ResponseField>

    <ResponseField name="campaign_id" type="integer">
      Bu müşterinin ait olduğu kampanyanın kimliği
    </ResponseField>

    <ResponseField name="phone_number" type="string">
      Müşterinin E.164 formatındaki telefon numarası
    </ResponseField>

    <ResponseField name="variables" type="object">
      Müşteriyle ilişkili değişkenler
    </ResponseField>

    <ResponseField name="status" type="string">
      Müşterinin durumu
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Müşterinin oluşturulma tarihi
    </ResponseField>

    <ResponseField name="updated_at" type="string">
      Müşterinin güncellenme tarihi
    </ResponseField>

    <ResponseField name="campaign" type="object">
      Kampanya bilgileri
      <Expandable title="kampanya özellikleri">
        <ResponseField name="id" type="integer">
          Kampanyanın kimliği
        </ResponseField>
        <ResponseField name="name" type="string">
          Kampanyanın adı
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="secondary_contacts" type="array">
      Bu müşteriyle ilişkili ikincil iletişim kişileri dizisi
      <Expandable title="secondary_contacts özellikleri">
        <ResponseField name="id" type="integer">
          İkincil iletişim kişisinin kimliği
        </ResponseField>
        <ResponseField name="phone_number" type="string">
          İkincil iletişim kişisinin E.164 formatındaki telefon numarası
        </ResponseField>
        <ResponseField name="variables" type="object">
          İkincil iletişim kişisiyle ilişkili değişkenler
        </ResponseField>
        <ResponseField name="status" type="string">
          İkincil iletişim kişisinin durumu
        </ResponseField>
        <ResponseField name="created_at" type="string">
          İkincil iletişim kişisinin oluşturulma tarihi
        </ResponseField>
        <ResponseField name="updated_at" type="string">
          İkincil iletişim kişisinin güncellenme tarihi
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="current_page" type="integer">
  Mevcut sayfa numarası
</ResponseField>

<ResponseField name="per_page" type="integer">
  Sayfa başına öğe sayısı
</ResponseField>

<ResponseField name="total" type="integer">
  Kriterlere uyan toplam müşteri sayısı
</ResponseField>

<ResponseField name="last_page" type="integer">
  Son sayfa numarası
</ResponseField>
